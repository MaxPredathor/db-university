SELECT `degrees`.`name` AS `nome_corso`, `students`.*
FROM `degrees`
JOIN `students` ON `students`.`degree_id` = `degrees`.`id`
WHERE `degrees`.`name` = 'Corso di Laurea in Economia';

2-
SELECT `departments`.`name` AS `nome`, `degrees`.*
FROM `degrees` 
JOIN `departments` ON `departments`.`id` = `degrees`.`department_id`
WHERE `departments`.`name` = 'Dipartimento di Neuroscienze' && `degrees`.`level` = 'magistrale'

3-
SELECT `teachers`.`name` AS `nome_docente`, `teachers`.`surname` AS `cognome_docente`, `teachers`.`id` AS `id_insegnante`, `courses`.* 
FROM `courses`
JOIN `course_teacher`
ON `course_teacher`.`course_id` = `courses`.`id`
JOIN `teachers`
ON `teachers`.`id` = `course_teacher`.`teacher_id`
WHERE `teachers`.`name` = 'Fulvio' && `teachers`.`surname` = 'Amato';

4-
SELECT `students`.`name` AS `nome_studente`, `students`.`surname` AS `cognome_studente`, `degrees`.*, `departments`.*
FROM `students`
JOIN `degrees`
ON `degrees`.`id` = `students`.`degree_id`
JOIN `departments`
ON `departments`.`id` = `degrees`.`department_id`
ORDER BY `nome_studente`, `cognome_studente`
